WINDOW 0 0 100 100
SET frame 0

LABEL loop
    # FIX: Use a 16x16 RECT to ensure all pixels from any character are erased.
    # We shift the X/Y to 45 so the 16x16 box is centered over the 50,50 draw point.
    RECT 45 45 16 16 0x222222
    
    CMP frame 0
    JE draw_pipe
    CMP frame 1
    JE draw_slash
    CMP frame 2
    JE draw_dash
    CMP frame 3
    JE draw_bslash
    
    SET frame 0
    GOTO loop

LABEL draw_pipe
    PRINT "|" 50 50 0x00FF00
    GOTO next_frame

LABEL draw_slash
    PRINT "/" 50 50 0x00FF00
    GOTO next_frame

LABEL draw_dash
    PRINT "-" 50 50 0x00FF00
    GOTO next_frame

LABEL draw_bslash
    PRINT "\" 50 50 0x00FF00
    GOTO next_frame

LABEL next_frame
    ADD frame 1
    SLEEP 100
    GOTO loop
