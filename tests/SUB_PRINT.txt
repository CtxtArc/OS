# Reactive Counter Test
WINDOW 0 0 400 200
SET count 0
SET last_key 0

LABEL draw_loop
    CLEAR
    RECT 0 0 400 30 0x333333
    PRINT "COUNTER APP - ARROWS TO CHANGE" 10 10 0xFFFF00
    
    # Print the current value (PRINT_NUM logic)
    PRINT "Current Value: " 50 80 0xFFFFFF
    PRINT count 180 80 0x00FF00

LABEL input_wait
    # Get user input into 'last_key'
    INPUT last_key
    
    # If no key (0), just loop back to draw
    CMP last_key 0
    JE input_wait

    # Check for RIGHT ARROW (14) -> ADD
    CMP last_key 14
    JE do_add

    # Check for LEFT ARROW (13) -> SUB
    CMP last_key 13
    JE do_sub

    # Check for CTRL+Q (17) -> EXIT
    CMP last_key 17
    JE do_exit

    GOTO input_wait

LABEL do_add
    ADD count 1
    GOTO draw_loop

LABEL do_sub
    # Safety: don't go below zero if you want
    CMP count 0
    JE draw_loop
    SUB count 1
    GOTO draw_loop

LABEL do_exit
    EXIT
