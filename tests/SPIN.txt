WINDOW 0 0 100 100
SET frame 0
LABEL loop
    CLEAR
    CMP frame 0
    JE draw_pipe
    CMP frame 1
    JE draw_slash
    CMP frame 2
    JE draw_dash
    CMP frame 3
    JE draw_bslash
    
    # If frame > 3, reset it
    SET frame 0
    GOTO loop

LABEL draw_pipe
    PRINT "|" 50 50 0x00FF00
    GOTO next

LABEL draw_slash
    PRINT "/" 50 50 0x00FF00
    GOTO next

LABEL draw_dash
    PRINT "-" 50 50 0x00FF00
    GOTO next

LABEL draw_bslash
    PRINT "\\" 50 50 0x00FF00
    GOTO next

LABEL next
    ADD frame 1
    SLEEP 100
    GOTO loop
