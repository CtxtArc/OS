WINDOW 0 0 500 400
SET index 0
SET draw_x 10
SET draw_y 10

# 1. Ensure a file was specified
GET_ARGC count
CMP count 0
JE no_args

# 2. Load the file from the argument string!
LOADFILE_ARG file_ptr file_size
CMP file_size 0
JE file_error

LABEL read_loop
    CMP index file_size
    JE cleanup

    READBYTE file_ptr index char_val
    
    # Skip null characters
    CMP char_val 0
    JE skip_char

    # Handle Newlines (ASCII 10)
    CMP char_val 10
    JE do_newline

    PRINT_CHAR char_val draw_x draw_y 0x00FF00

    ADD draw_x 8
    CMP draw_x 480
    JL next_char
    
LABEL do_newline
    SET draw_x 10
    ADD draw_y 12

LABEL next_char
    ADD index 1
    GOTO read_loop

LABEL skip_char
    ADD index 1
    GOTO read_loop

LABEL cleanup
    FREE file_ptr

LABEL wait_exit
    PRINT "(Press Q to close)" 10 380 0x555555
    GETKEY k
    CMP k 113
    JE quit
    SLEEP 100
    GOTO wait_exit

LABEL no_args
    PRINT "ERROR: Provide a filename" 10 10 0xFF0000
    GOTO wait_exit

LABEL file_error
    PRINT "ERROR: File empty or not found" 10 10 0xFF0000
    GOTO wait_exit

LABEL quit
    EXIT
